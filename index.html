// === PASTE YOUR FIREBASE CONFIG HERE ===  
const firebaseConfig = {  
apiKey: "YOUR_API_KEY_HERE",  
authDomain: "YOUR_PROJECT.firebaseapp.com",  
projectId: "YOUR_PROJECT",  
storageBucket: "YOUR_PROJECT.appspot.com",  
messagingSenderId: "YOUR_SENDER_ID",  
appId: "YOUR_APP_ID"  
};  
// Initialize Firebase  
firebase.initializeApp(firebaseConfig);  
const auth = firebase.auth();  
const db = firebase.firestore();  
let currentUser = null;  
let currentWarehouse = "Main Warehouse";  
// Auth functions  
function register() {  
const email = document.getElementById('email').value;  
const password = document.getElementById('password').value;  
auth.createUserWithEmailAndPassword(email, password)  
.then(() => { alert("Account created! Now login."); })  
.catch(err => document.getElementById('authMessage').innerText = err.message);  
}
function login() {  
const email = document.getElementById('email').value;  
const password = document.getElementById('password').value;  
auth.signInWithEmailAndPassword(email, password)  
.then(userCredential => {  
currentUser = userCredential.user;  
document.getElementById('loginScreen').style.display = 'none';  
document.getElementById('appScreen').classList.add('active');  
initApp();  
})  
.catch(err => document.getElementById('authMessage').innerText = err.message);  
}
function logout() {  
auth.signOut().then(() => location.reload());  
}
// Show tabs  
function showTab(tabId) {  
document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));  
document.getElementById(tabId).classList.add('active');  
if (tabId === 'dashboard') loadItems();  
}
// Initialize after login  
function initApp() {  
// Warehouses (you can expand this)  
const warehouses = ["Main Warehouse", "Bangalore Warehouse", "Delhi Warehouse"];  
const whSelect = document.getElementById('warehouseSelect');  
const itemWh = document.getElementById('itemWarehouse');  
warehouses.forEach(w => {  
const opt = document.createElement('option');  
opt.value = w; opt.textContent = w;  
whSelect.appendChild(opt.cloneNode(true));  
itemWh.appendChild(opt);  
});  
loadItems();  
}
// Realtime load items  
function loadItems() {  
currentWarehouse = document.getElementById('warehouseSelect').value || "Main  
Warehouse";  
const tbody = document.getElementById('stockTable').querySelector('tbody');  
tbody.innerHTML = '';  
db.collection('items')  
.where('warehouse', '==', currentWarehouse)  
.onSnapshot(snapshot => {  
tbody.innerHTML = '';  
let lowStock = [];  
snapshot.forEach(doc => {  
const data = doc.data();  
const row = document.createElement('tr');  
row.innerHTML = `  
<td>${data.name}</td>  
<td>${data.batch || '-'}</td>  
<td>${data.expiry || '-'}</td>  
<td>${data.quantity}</td>  
<td>${data.warehouse}</td>  
`;  
tbody.appendChild(row);  
if (data.quantity < 10) lowStock.push(data);  
});  
// Replenishment table  
const lowBody = document.getElementById('replenishTable').querySelector('tbody');  
lowBody.innerHTML = '';  
lowStock.forEach(item => {  
const row = document.createElement('tr');  
row.innerHTML = `<td>\( {item.name}</td><td> \){item.quantity}</td>`;  
lowBody.appendChild(row);  
});  
});  
}
// Add new item  
function addItem() {  
const name = document.getElementById('itemName').value;  
const batch = document.getElementById('batch').value;  
const expiry = document.getElementById('expiry').value;  
const qty = parseInt(document.getElementById('qty').value) || 0;  
const warehouse = document.getElementById('itemWarehouse').value;  
if (!name) return alert("Item name required");  
db.collection('items').add({  
name: name,  
batch: batch,  
expiry: expiry,  
quantity: qty,  
warehouse: warehouse,  
createdBy: currentUser.email,  
timestamp: firebase.firestore.FieldValue.serverTimestamp()  
}).then(() => {  
alert("Item added!");  
loadItems();  
});  
}
// Stock adjustment (supports + for purchase/receive, - for sales/return/damage)  
function adjustStock() {  
const itemSelect = document.getElementById('adjustItem');  
// For simplicity we reload items into select each time (you can improve later)  
// In real use, populate select from items collection  
const qtyChange = parseInt(document.getElementById('adjustQty').value);  
const note = document.getElementById('adjustNote').value || "Adjustment";  
if (!qtyChange) return alert("Enter quantity");  
// For demo, we add a simple transaction log instead of updating existing item  
db.collection('adjustments').add({  
type: qtyChange > 0 ? "Purchase/Receive" : "Return/Damage/Sales",  
quantity: qtyChange,  
note: note,  
warehouse: currentWarehouse,  
user: currentUser.email,  
timestamp: firebase.firestore.FieldValue.serverTimestamp()  
}).then(() => {  
alert("Adjustment recorded! (Note: Full item update logic can be added later)");  
loadItems();  
});  
}
